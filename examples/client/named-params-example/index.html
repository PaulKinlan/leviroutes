Code Example

// Named parameters are defined with a colon
app.get("/user/:username", function(req) {
  console.log("Username:", req.params.username);
});

// Multiple parameters
app.get("/user/:username/post/:postId", function(req) {
  console.log("User:", req.params.username);
  console.log("Post ID:", req.params.postId);
});

// Parameters with format
app.get("/:category.:format", function(req) {
  console.log("Category:", req.params.category);
  console.log("Format:", req.params.format);
});
